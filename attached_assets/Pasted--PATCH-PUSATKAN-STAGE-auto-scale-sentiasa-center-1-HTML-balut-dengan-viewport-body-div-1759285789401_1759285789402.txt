2) CSS (center dengan CSS Grid, tiada translate)
html, body {

  background: #000;           /* border hitam */
  overflow: hidden;
}

/* Viewport penuh skrin & CENTER anak */
.viewport {
  width: 100vw;
  height: 100vh;
  display: grid;
  place-items: center;        /* <â€” kunci: center H & V */
}

/* Kanvas tetap 1920x1080, hanya skala berubah */
#stage {
  width: 1920px;
  height: 1080px;
  background: #0f1d3a;        /* warna paparan */
  color: #fff;
  overflow: hidden;
  transform: scale(1);
  transform-origin: center center;   /* <â€” scale dari tengah */
}

/* Jangan biar anak paksa overflow */
#stage, #stage * {
  min-width: 0; min-height: 0;
}

3) JS (kira skala, kekal center)
<script>
  (function () {
    const W = 1920, H = 1080;
    const stage = document.getElementById('stage');
    const viewport = document.getElementById('viewport');

    function fit() {
      const s = Math.min(viewport.clientWidth / W, viewport.clientHeight / H);
      stage.style.transform = `scale(${s})`; // center dijaga oleh .viewport
    }

    addEventListener('resize', fit);
    addEventListener('orientationchange', fit);
    fit();
  })();
</script>
